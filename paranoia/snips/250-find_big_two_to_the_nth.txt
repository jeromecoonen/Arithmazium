def find_big_2_to_nth():
    """Compute 2**n satisfying |((2**n + 1)-2**n)-1| >= 1."""
    x = ONE
    while True:
        x = x + x
        y = x + ONE
        z = y - x
        y = z - ONE
        if MINUS_ONE + fabs(y) >= ZERO: break
    # ... now x is just big enough that |((x+1)-x)-1| >= 1 ...
    return x
