    q = floor(ONE_HALF - log(C) / log(A))   # C is a power of B and thus A
    for z in [A, ONE_OVER_A]:

        test_power_result(x, z, q)
        q = -q
        x = C
        test_power_result(x, z, q)
    print_if_err_cnt_positive()
